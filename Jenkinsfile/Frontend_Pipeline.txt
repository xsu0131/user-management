pipeline {
    agent any

    tools {
        nodejs 'NodeJS-20'
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/xsu0131/user-management.git'
            }
        }

        stage('Install Dependencies') {
            steps {
                dir('frontend') {
                    sh '''
                        rm -rf node_modules package-lock.json
                        npm cache clean --force
                        npm install
                    '''
                }
            }
        }

        stage('Build Angular') {
            steps {
                dir('frontend') {
                    sh 'npm run build -- --configuration production'
                }
            }
        }

        stage('Deploy Frontend') {
            steps {
                sh """
                    ansible-playbook \
                    -i infra/ansible/inventory/hosts \
                    infra/ansible/playbooks/deploy_frontend.yml
                """
            }
        }

        stage('Health Check') {
            steps {
                script {
                    sleep 5
                    def frontendIp = sh(
                        script: "terraform -chdir=infra/terraform output -raw frontend_ip",
                        returnStdout: true
                    ).trim()

                    sh "curl -f http://${frontendIp}"
                }
            }
        }
    }

    post {
        success {
            echo "✅ Frontend deployed successfully"
        }
        failure {
            echo "❌ Frontend deployment failed"
        }
    }
}
