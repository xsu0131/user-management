pipeline {
    agent any

    tools {
        nodejs 'NodeJS-20'
    }

    environment {
        SSH_CREDENTIAL = 'ec2-ssh-key'   // Must match Jenkins credential ID
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/xsu0131/user-management.git'
            }
        }

        stage('Terraform Init') {
            steps {
                sh 'terraform -chdir=infra/terraform init'
            }
        }

        stage('Install Dependencies') {
            steps {
                dir('frontend') {
                    sh '''
                        rm -rf node_modules package-lock.json
                        npm cache clean --force
                        npm install
                    '''
                }
            }
        }

        stage('Build Angular') {
            steps {
                dir('frontend') {
                    sh 'npm run build'
                }
            }
        }

        stage('Get Server IPs') {
            steps {
                script {
                    env.FRONTEND_IP = sh(
                        script: 'terraform -chdir=infra/terraform output -raw frontend_ip',
                        returnStdout: true
                    ).trim()

                    env.BACKEND_IP = sh(
                        script: 'terraform -chdir=infra/terraform output -raw backend_ip',
                        returnStdout: true
                    ).trim()

                    echo "Frontend IP: ${env.FRONTEND_IP}"
                    echo "Backend IP: ${env.BACKEND_IP}"
                }
            }
        }

        stage('Deploy') {
            steps {
                sshagent(credentials: ['ec2-ssh-key']) {
                    sh """
                        # Create temp dir on frontend
                        ssh -o StrictHostKeyChecking=no ubuntu@${env.FRONTEND_IP} \
                        "sudo mkdir -p /tmp/angular-dist && sudo chown ubuntu:ubuntu /tmp/angular-dist"

                        # Copy Angular build
                        scp -o StrictHostKeyChecking=no -r \
                        frontend/dist/user-management/* \
                        ubuntu@${env.FRONTEND_IP}:/tmp/angular-dist/

                        # ---------- CREATE FULL INVENTORY ----------
                        echo "[frontend]" > inventory.ini
                        echo "${env.FRONTEND_IP}" >> inventory.ini
                        echo "" >> inventory.ini

                        echo "[backend]" >> inventory.ini
                        echo "${env.BACKEND_IP}" >> inventory.ini
                        echo "" >> inventory.ini

                        echo "[all:vars]" >> inventory.ini
                        echo "ansible_user=ubuntu" >> inventory.ini
                        echo "ansible_ssh_private_key_file=/home/ubuntu/.ssh/project1u.pem" >> inventory.ini
                        echo "ansible_python_interpreter=/usr/bin/python3" >> inventory.ini

                        echo "====== Generated Inventory ======"
                        cat inventory.ini
                        echo "=================================="

                        # Run Ansible
                        ansible-playbook -i inventory.ini \
                        infra/ansible/playbooks/deploy_frontend.yml
                    """
                }
            }
        }

        stage('Health Check') {
            steps {
                sh '''
                    sleep 5
                    curl -f http://${FRONTEND_IP}
                '''
            }
        }
    }

    post {
        success {
            echo " Frontend deployed successfully!"
            echo " Access at: http://${FRONTEND_IP}"
        }
        failure {
            echo " Frontend deployment failed!"
        }
    }
}
