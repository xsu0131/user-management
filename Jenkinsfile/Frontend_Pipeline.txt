pipeline {
    agent any

    tools {
        nodejs 'NodeJS-20'
    }

    environment {
        SSH_CREDENTIAL = 'ec2-ssh-key'
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/xsu0131/user-management.git'
            }
        }

        stage('Terraform Init') {
            steps {
                sh 'terraform -chdir=infra/terraform init'
            }
        }

        stage('Install Dependencies') {
            steps {
                dir('frontend') {
                    sh '''
                        rm -rf node_modules package-lock.json
                        npm cache clean --force
                        npm install
                    '''
                }
            }
        }

        stage('Build') {
            steps {
                dir('frontend') {
                    sh 'npm run build'
                }
            }
        }

        stage('Get Frontend IP') {
            steps {
                script {
                    env.FRONTEND_IP = sh(
                        script: 'terraform -chdir=infra/terraform output -raw frontend_ip',
                        returnStdout: true
                    ).trim()
                    echo "Frontend IP: ${env.FRONTEND_IP}"
                }
            }
        }

        stage('Deploy') {
            steps {
                sshagent (credentials: [env.SSH_CREDENTIAL]) {
                    sh """
                        # Create temp directory on EC2
                        ssh -o StrictHostKeyChecking=no ubuntu@${env.FRONTEND_IP} \
                            'sudo mkdir -p /tmp/angular-dist && sudo chown ubuntu:ubuntu /tmp/angular-dist'

                        # Copy Angular build output
                        scp -o StrictHostKeyChecking=no \
                            -r frontend/dist/user-management/* \
                            ubuntu@${env.FRONTEND_IP}:/tmp/angular-dist/

                        # Run Ansible deployment
                        ansible-playbook \
                            infra/ansible/playbooks/deploy_frontend.yml \
                            -e "frontend_ip=${env.FRONTEND_IP}"
                    """
                }
            }
        }

        stage('Health Check') {
            steps {
                sh 'sleep 5'
                sh 'curl -f http://${FRONTEND_IP}'
            }
        }
    }

    post {
        success {
            echo " Frontend deployed successfully!"
            echo " Access at: http://${FRONTEND_IP}"
        }
        failure {
            echo " Frontend deployment failed!"
        }
    }
}
